package com.ucp.tetris;

public class PieceT extends Piece{

    public PieceT(){
        piezaInicial();
        getWidth();
        getHeight();
    }

    public void piezaInicial(){
        matriz[0][0]= 1;
        matriz[0][1]= 1;
        matriz[0][2]= 1;
        matriz[1][1]= 1;
        }

        @Override
        public void rotarDerecha() {
            if (matriz[0][0] == 1 && matriz[0][1] == 1 && matriz[0][2] == 1 && matriz[1][1] == 1) { //rotar derecha T inicial
                limpiarMatriz();
                matriz[0][1] = 1;
                matriz[1][1] = 1;
                matriz[1][0] = 1;
                matriz[2][1] = 1;
                setWidth(2); 
                setHeight(3); 
            } else if (matriz[0][1] == 1 && matriz[1][1] == 1 && matriz[1][0] == 1 && matriz[2][1] == 1) { //rotar derecha 2 veces 
                limpiarMatriz();
                matriz[1][0] = 1;
                matriz[1][1] = 1;
                matriz[1][2] = 1;
                matriz[0][1] = 1;
                setWidth(3); 
                setHeight(2); 
            } else if (matriz[1][0] == 1 && matriz[1][1] == 1 && matriz[1][2] == 1 && matriz[0][1] == 1) {//rotar derecha 3 veces
                limpiarMatriz();
                matriz[0][1] = 1;
                matriz[1][1] = 1;
                matriz[1][2] = 1;
                matriz[2][1] = 1;
                setWidth(2); 
                setHeight(3); 
            } else if (matriz[0][1] == 1 && matriz[1][1] == 1 && matriz[2][1] == 1 && matriz[1][2] == 1) {//rotar derecha 4 veces
                limpiarMatriz();
                 matriz[0][0]= 1;
                 matriz[0][1]= 1;
                 matriz[0][2]= 1;
                 matriz[1][1]= 1;
                setWidth(3); 
                setHeight(2); 
            }// rotaciones cuando se mueven las columnas: T inicial
            else if(matriz[0][1] == 1 && matriz[0][2] == 1 && matriz[0][3] == 1 && matriz[1][2] == 1){
                limpiarMatriz();
                matriz[0][2] = 1;
                matriz[1][2] = 1;
                matriz[1][1] = 1;
                matriz[2][2] = 1;
            }else if (matriz[0][2] == 1 && matriz[1][2] == 1 && matriz[1][1] == 1 && matriz[2][2] == 1) { //rotar derecha 2 veces 
                limpiarMatriz();
                matriz[1][1] = 1;
                matriz[1][2] = 1;
                matriz[1][3] = 1;
                matriz[0][2] = 1;
                setWidth(3); 
                setHeight(2); 
            } else if (matriz[1][1] == 1 && matriz[1][2] == 1 && matriz[1][3] == 1 && matriz[0][2] == 1) {//rotar derecha 3 veces
                limpiarMatriz();
                matriz[0][2] = 1;
                matriz[1][2] = 1;
                matriz[1][3] = 1;
                matriz[2][2] = 1;
                setWidth(2); 
                setHeight(3); 
            } else if (matriz[0][2] == 1 && matriz[1][2] == 1 && matriz[2][2] == 1 && matriz[1][3] == 1) {//rotar derecha 4 veces
                limpiarMatriz();
                 matriz[0][1]= 1;
                 matriz[0][2]= 1;
                 matriz[0][3]= 1;
                 matriz[1][2]= 1;
                setWidth(3); 
                setHeight(2); 
            }//T rotada derecha dos veces
            else if(matriz[0][3] == 1 && matriz[1][3] == 1 && matriz[2][3] == 1 && matriz[1][2] == 1){//rotar derecha 1 vez
                limpiarMatriz();
                matriz[1][1] = 1;
                matriz[1][2] = 1;
                matriz[1][3] = 1;
                matriz[0][2] = 1;
            }else if (matriz[0][0] == 1 && matriz[1][0] == 1 && matriz[2][0] == 1 && matriz[1][1] == 1) { //rotar derecha 2 veces 
                limpiarMatriz();
                matriz[0][0]= 1;
                matriz[0][1]= 1;
                matriz[0][2]= 1;
                matriz[1][1]= 1;
    }
}
