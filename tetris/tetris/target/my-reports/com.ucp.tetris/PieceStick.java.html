package com.ucp.tetris;

public class PieceStick extends Piece{
    public PieceStick(){
    piezaInicial();
    getWidth();
    getHeight();
    }

    public void piezaInicial(){
    matriz[0][0]= 1;
    matriz[0][1]= 1;
    matriz[0][2]= 1;
    matriz[0][3]= 1;
    }

    @Override
        public void rotarDerecha(){
            if (matriz[0][0] == 1 && matriz[0][1] == 1 && matriz[0][2] == 1 && matriz[0][3] == 1){//rotar derecha Stick
            limpiarMatriz();    
            matriz[0][2]= 1;
            matriz[1][2]= 1;
            matriz[2][2]= 1;
            matriz[3][2]= 1;
            getWidth1();
            }else if(matriz[0][2] == 1 && matriz[1][2] == 1 && matriz[2][2] == 1 && matriz[3][2] == 1){//rotar derecha Stick rotado
            limpiarMatriz();
            matriz[0][0]= 1;
            matriz[0][1]= 1;
            matriz[0][2]= 1;
            matriz[0][3]= 1;
            getWidth();
            }//Rotaciones cuando se mueven las columnas 
            else if(matriz[0][0] == 1 && matriz[1][0] == 1 && matriz[2][0] == 1 && matriz[3][0] == 1){
            limpiarMatriz();
            matriz[0][0]= 1;
            matriz[0][1]= 1;
            matriz[0][2]= 1;
            matriz[0][3]= 1;
            getWidth();
    }else if(matriz[0][3] == 1 && matriz[1][3] == 1 && matriz[2][3] == 1 && matriz[3][3] == 1){
            limpiarMatriz();
            matriz[0][0]= 1;
            matriz[0][1]= 1;
            matriz[0][2]= 1;
            matriz[0][3]= 1;
            getWidth();
    }
}

    @Override
        public void rotarIzquierda(){
            if (matriz[0][0] == 1 && matriz[0][1] == 1 && matriz[0][2] == 1 && matriz[0][3] == 1){
            limpiarMatriz();    
            matriz[0][1]= 1;
            matriz[1][1]= 1;
            matriz[2][1]= 1;
            matriz[3][1]= 1;
            getWidth1();

            limpiarMatriz();
            matriz[0][0]= 1;

            matriz[0][3]= 1;

            limpiarMatriz();

}

    @Override
    public int getWidth(){

        return 4; 
}

    public int getWidth1(){

        return 1; 
}

        return 1;

    public void moverColumnaDerecha(){
        if(canMoveMatriz() == true){
            if(matriz[0][2] == 1 && matriz[1][2] == 1 && matriz[2][2] == 1 && matriz[3][2] == 1){
            limpiarMatriz();
            matriz[0][3]= 1;
            matriz[1][3]= 1;

            matriz[3][3]= 1;

            else if(matriz[0][1] == 1 && matriz[1][1] == 1 && matriz[2][1] == 1 && matriz[3][1] == 1){

}
    }
