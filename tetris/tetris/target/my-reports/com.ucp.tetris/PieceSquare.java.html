    matriz[0][0]= 1;
    matriz[0][1]= 1;
    matriz[1][0]= 1;
    matriz[1][1]= 1;
    }

    public void rotarIzquierda(){

    }

    public void rotarDerecha(){

    }

    @Override
    public int getWidth() {

}
    @Override
    public int getHeight() {

    return 2; 
}
@Override
public void moverColumnaDerecha(){
        if(canMoveMatriz() == true){
            if(matriz[0][0] == 1 && matriz[0][1] == 1 && matriz[1][0] == 1 && matriz[1][1] == 1){
            limpiarMatriz();
            matriz[0][1]= 1;
            matriz[0][2]= 1;
            matriz[1][1]= 1;
            matriz[1][2]= 1;
            }
            else if(matriz[0][1] == 1 && matriz[0][2] == 1 && matriz[1][1] == 1 && matriz[1][2] == 1){
            limpiarMatriz();
            matriz[0][2]= 1;
            matriz[0][3]= 1;
            matriz[1][2]= 1;
            matriz[1][3]= 1;
    }
}
    }
